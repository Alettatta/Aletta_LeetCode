# Problem Description
给定一个 m x n 的整数矩阵 matrix，如果矩阵中的某个元素为 0，则将其所在的整行和整列的所有元素都置为 0。

请使用原地算法完成这一修改。


# Code

## LC version

```python
class Solution:
    def setZeroes(self, matrix: List[List[int]]) -> None:
        # matrix[i][j] -> matrix[i][…],matrix[…][j]
        # 记录哪些行、哪些列需要置0 -> 原地操作：matrix[0][j] 标记第 j 列是否需要置零，matrix[i][0] 标记第 i 行是否需要置零
        # 但是第一行本身有数据，不能被覆盖

        m,n = len(matrix), len(matrix[0])
        # 判断第一行和第一列是否本身就有0
        first_row_zero = any(matrix[0][j] == 0 for j in range(n))
        first_col_zero = any(matrix[i][0] == 0 for i in range(m))

        # 这下可以使用第一行/列标记了
        for i in range(1, m):
            for j in range(1, n): # 操作的对象是 matrix[1:][1:],也就是除了第一行第一列之外的主体区域
                if matrix[i][j] == 0:
                    matrix[i][0] = 0
                    matrix[0][j] = 0
        # 根据标记置0
        for i in range(1, m):
            for j in range(1, n): # 操作的对象是 matrix[1:][1:],也就是除了第一行第一列之外的主体区域
                if matrix[i][0] == 0 or matrix[0][j] == 0:
                    matrix[i][j] = 0

        # 最后处理第一行/列
        if first_row_zero is True:
            for j in range(n):
                matrix[0][j] = 0
        if first_col_zero is True:
            for i in range(m):
                matrix[i][0] = 0
```

## ACM version

**ACM 模式的注意点：**

- 需要 import 完整的类（包括 sys、typing 等）
- 数据在标准输入流 stdin 中，全部是原始的文本字符串
- 必须用 print() 手动将结果写到标准输出流 stdout
- 需要写 while 或 for line in sys.stdin 循环处理，直到文件结束（EOF）

```python
import sys
from typing import List

class Solution:
    def setZeroes(self, matrix: List[List[int]]) -> None:
        # matrix[i][j] -> matrix[i][…],matrix[…][j]
        # 记录哪些行、哪些列需要置0 -> 原地操作：matrix[0][j] 标记第 j 列是否需要置零，matrix[i][0] 标记第 i 行是否需要置零
        # 但是第一行本身有数据，不能被覆盖

        m,n = len(matrix), len(matrix[0])
        # 判断第一行和第一列是否本身就有0
        first_row_zero = any(matrix[0][j] == 0 for j in range(n))
        first_col_zero = any(matrix[i][0] == 0 for i in range(m))

        # 这下可以使用第一行/列标记了
        for i in range(1, m):
            for j in range(1, n): # 操作的对象是 matrix[1:][1:],也就是除了第一行第一列之外的主体区域
                if matrix[i][j] == 0:
                    matrix[i][0] = 0
                    matrix[0][j] = 0
        # 根据标记置0
        for i in range(1, m):
            for j in range(1, n): # 操作的对象是 matrix[1:][1:],也就是除了第一行第一列之外的主体区域
                if matrix[i][0] == 0 or matrix[0][j] == 0:
                    matrix[i][j] = 0

        # 最后处理第一行/列
        if first_row_zero is True:
            for j in range(n):
                matrix[0][j] = 0
        if first_col_zero is True:
            for i in range(m):
                matrix[i][0] = 0

state = {"m": 0, "n": 0, "rows_left": 0, "matrix": []}
for line in sys.stdin:
    parts = line.split()
    if not parts:
        continue
    if state["rows_left"] == 0:
        state["m"], state["n"] = int(parts[0]), int(parts[1])
        state["rows_left"] = state["m"]
        state["matrix"] = []
    else:
        state["matrix"].append(list(map(int, parts)))
        state["rows_left"] -= 1
        if state["rows_left"] == 0:
            Solution().setZeroes(state["matrix"])
            for row in state["matrix"]:
                print(' '.join(map(str, row)))
```
